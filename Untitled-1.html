function toThaiDateString(date) { let monthNames = [ "มกราคม", "กุมภาพันธ์", "มีนาคม", "เมษายน", "พฤษภาคม", "มิถุนายน", "กรกฎาคม", "สิงหาคม.", "กันยายน", "ตุลาคม", "พฤศจิกายน", "ธันวาคม" ]; let year = date.getFullYear() + 543; let month = monthNames[date.getMonth()]; let numOfDay = date.getDate(); let hour = date.getHours().toString().padStart(2, "0"); let minutes = date.getMinutes().toString().padStart(2, "0"); let second = date.getSeconds().toString().padStart(2, "0"); return `${numOfDay} ${month} ${year} ` + `${hour}:${minutes}:${second} น.`; }
